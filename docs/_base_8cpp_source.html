136  state_ = State::noBattery;
137  else if (state == SDL_POWERSTATE_CHARGING)
138  state_ = State::charging;
139  else if (state == SDL_POWERSTATE_CHARGED)
140  state_ = State::charged;
141  else if (state == SDL_POWERSTATE_UNKNOWN)
